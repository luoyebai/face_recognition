89 template <typename CallBack>

91 size_t fps = 10) const {
92 auto file_path = config_path_ + file_name;
93 if (!std::filesystem::exists(file_path))
94 return false;
95 std::thread([file_path, f, fps] {
96 auto last_time = std::filesystem::last_write_time(file_path);
97 while (true) {
98 std::this_thread::sleep_for(
99 std::chrono::milliseconds(1000 / fps));
100 auto now_time = std::filesystem::last_write_time(file_path);
101 // std::cou << last_time << "\t" << now_time << "\n";
102 if (last_time.time_since_epoch() == now_time.time_since_epoch())
103 continue;
104 last_time = std::filesystem::last_write_time(file_path);
105 f();
106 }
107 }).detach();
